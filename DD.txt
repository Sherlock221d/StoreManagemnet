using System;
using System.Collections.Generic;
using System.Configuration;
using System.Data;
using System.Data.SqlClient;
using System.IO;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;

namespace StoreManagementSystem
{
    public partial class ItemEntryPage : System.Web.UI.Page
    {
        private string strConnectionString = ConfigurationManager.ConnectionStrings["DBconnection"].ConnectionString;
        private SqlCommand _sqlCommand;
        public void CreateConnection()
        {
            SqlConnection _sqlConnection = new SqlConnection(strConnectionString);
            _sqlCommand = new SqlCommand();
            _sqlCommand.Connection = _sqlConnection;
        }
        public void OpenConnection()
        {
            _sqlCommand.Connection.Open();
        }
        public void CloseConnection()
        {
            _sqlCommand.Connection.Close();
        }
        protected void Page_Load(object sender, EventArgs e)
        {
            if (!IsPostBack)
            {
                btnUpdate.Visible = false;
                GetItemEntryMasterList();
            }
        }
        private static void ShowAlertMessage(string error)
        {
            System.Web.UI.Page page = System.Web.HttpContext.Current.Handler as System.Web.UI.Page;
            if (page != null)
            {
                error = error.Replace("'", "\'");
                System.Web.UI.ScriptManager.RegisterStartupScript(page, page.GetType(), "err_msg", "alert('" + error + "');", true);
            }
        }
        private void LogError(Exception ex)
        {
            string logFilePath = Server.MapPath("~/ErrorLog.txt");
            using (StreamWriter writer = new StreamWriter(logFilePath, true))
            {
                writer.WriteLine("----------------------------------------------------------------------------------------------------------------------------");
                writer.WriteLine("Error Logged at " + DateTime.Now);
                writer.WriteLine("Error Message: " + ex.Message);
                writer.WriteLine("Stack Trace: " + ex.StackTrace);
            }
        }
        public void GetItemEntryMasterList()
        {
            try
            {
                CreateConnection();
                OpenConnection();
                _sqlCommand.CommandText = "Sp_GetUserDetails";
                _sqlCommand.CommandType = CommandType.StoredProcedure;
                SqlDataReader dr = _sqlCommand.ExecuteReader();
                DataTable dt = new DataTable();
                dt.Load(dr);

                grdItemEntry.EmptyDataText = "No Records Found";
                grdItemEntry.DataSource = dt;
                grdItemEntry.DataBind();
            }
            catch (Exception ex)
            {
                ShowAlertMessage("Check your input data");
            }
            finally
            {
              
            }

        }

        protected void btnSubmit_Click(object sender, EventArgs e)
        {
            try
            {
                CreateConnection();
                OpenConnection();

                if (IsDuplicateItem(txtItemName.Text))
                {
                    ShowAlertMessage("Item with the same name already exists. Please enter a different ItemName.");
                    return;
                }

                _sqlCommand = new SqlCommand("Sp_ItemEntryInsert", _sqlCommand.Connection);
                _sqlCommand.CommandType = CommandType.StoredProcedure;
                _sqlCommand.Parameters.AddWithValue("@ItemName", txtItemName.Text);
                _sqlCommand.Parameters.AddWithValue("@Rate", txtRates.Text);
                _sqlCommand.Parameters.AddWithValue("@BalanceQty", txtBalalnceQty.Text);
                _sqlCommand.Parameters.AddWithValue("@Category", ddlCategory.Text);
           

                int result = _sqlCommand.ExecuteNonQuery();

                if (result > 0)
                {
                    ShowAlertMessage("Record inserted  successfully");
                    GetItemEntryMasterList();
                    ClearFormFields();
                }
                else
                {
                    ShowAlertMessage("The entered data already exists");
                }

            }
            catch (Exception ex)
            {
                LogError(ex);
            }
            finally
            {

            }
        }
        private void ClearFormFields()
        {

            ddlCategory.SelectedIndex = 0;
            txtItemName.Text = string.Empty;
            txtRates.Text = string.Empty;
            txtBalalnceQty.Text = string.Empty;
           
        }
        private bool IsDuplicateItem(string itemName)
        {
            try
            {
                CreateConnection();
                OpenConnection();
                _sqlCommand = new SqlCommand("Sp_CheckDuplicateItem", _sqlCommand.Connection);
                _sqlCommand.CommandType = CommandType.StoredProcedure;
                _sqlCommand.Parameters.AddWithValue("@ItemName", itemName);
                int count = Convert.ToInt32(_sqlCommand.ExecuteScalar());
                return count > 0;
            }
            catch (Exception ex)
            {
               
                LogError(ex);
                return false; 
            }
            finally
            {
            }
        }

        string Id;
        protected void lnkupdate_Click(object sender, EventArgs e)
        {
            btnSubmit.Visible = false;
            btnUpdate.Visible = true;
            Id = (sender as LinkButton).CommandArgument;
            Session["UId"] = Id;
            UsersDatabyid();
        }
        public void UsersDatabyid()
        {
            
            try
            {
                int uid = Convert.ToInt32(Session["UId"]);
                CreateConnection();
                OpenConnection();
                _sqlCommand.CommandText = "SP_GetItemEntryById";
                _sqlCommand.CommandType = CommandType.StoredProcedure;
                _sqlCommand.Parameters.AddWithValue("@Id", uid);
                SqlDataReader dr = _sqlCommand.ExecuteReader();
                DataTable dt = new DataTable();
                dt.Load(dr);

                string Name = dt.Rows[0]["Item_Name"].ToString();
                txtItemName.Text = Name;

                string Rates = dt.Rows[0]["Rate"].ToString();
                txtRates.Text = Rates;

                string Balance_Quantity = dt.Rows[0]["Balance_Quantity"].ToString();
                txtBalalnceQty.Text = Balance_Quantity;

                string Category = dt.Rows[0]["Category"].ToString();
                ddlCategory.SelectedItem.Text = Category;



            }
            catch (Exception ex)
            {
                LogError(ex);
                ShowAlertMessage("Something went wrong....");
            }
            finally
            {

            }
        }

        protected void btnUpdate_Click(object sender, EventArgs e)
        {
            int uuid = Convert.ToInt32(Session["UId"]);
            try
            {
                CreateConnection();
                OpenConnection();
                _sqlCommand = new SqlCommand("Sp_ItemEntryUpdate", _sqlCommand.Connection);
                _sqlCommand.CommandType = CommandType.StoredProcedure;
                _sqlCommand.Parameters.AddWithValue("@Id", uuid);
                _sqlCommand.Parameters.AddWithValue("@ItemName", txtItemName.Text);
                _sqlCommand.Parameters.AddWithValue("@Rate", txtRates.Text);
                _sqlCommand.Parameters.AddWithValue("@BalanceQty", txtBalalnceQty.Text);
                _sqlCommand.Parameters.AddWithValue("@Category", ddlCategory.Text);


                int result = _sqlCommand.ExecuteNonQuery();

                if (result > 0)
                {
                    ShowAlertMessage("Record Updated  successfully");
                    GetItemEntryMasterList();
                    ClearFormFields();
                }
                else
                {
                    ShowAlertMessage("The entered data already exists");
                }

            }
            catch (Exception ex)
            {
                LogError(ex);
            }
            finally
            {

            }
        }

        protected void btnSearch_Click(object sender, EventArgs e)
        {
            try
            {
               
                
                string search = txtSearch.Text.Trim();

                if (!string.IsNullOrEmpty(search))
                {
                    CreateConnection();
                    OpenConnection();

                    _sqlCommand.CommandText = "Sp_SearchItemEntry";
                    _sqlCommand.CommandType = CommandType.StoredProcedure;
                    _sqlCommand.Parameters.Clear();
                    _sqlCommand.Parameters.AddWithValue("@txtsearch", search);

                    SqlDataReader reader = _sqlCommand.ExecuteReader();

                    if (reader.HasRows)
                    {
                        DataTable dt = new DataTable();
                        dt.Load(reader);

                        grdItemEntry.DataSource = dt;
                        grdItemEntry.DataBind();
                    }
                    else
                    {
                        grdItemEntry.DataSource = null;
                        grdItemEntry.DataBind();
                    }
                }
                else
                {
                    grdItemEntry.DataSource = null;
                    grdItemEntry.DataBind();
                }
            }
            catch (Exception ex)
            {
                
                LogError(ex);
                ShowAlertMessage("Something went wrong while searching.");
            }
            finally
            {
               
            }
        }

        protected void btnBack_Click(object sender, EventArgs e)
        {
            Response.Redirect("ItemEntryPage.aspx");
        }
    }
}

using System.IO;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;

namespace StoreManagementSystem
{
    public partial class To_FromReports : System.Web.UI.Page
    {
        private string strConnectionString = ConfigurationManager.ConnectionStrings["DBconnection"].ConnectionString;
        private SqlCommand _sqlCommand;
        public void CreateConnection()
        {
            SqlConnection _sqlConnection = new SqlConnection(strConnectionString);
            _sqlCommand = new SqlCommand();
            _sqlCommand.Connection = _sqlConnection;
        }
        public void OpenConnection()
        {
            _sqlCommand.Connection.Open();
        }
        public void CloseConnection()
        {
            _sqlCommand.Connection.Close();
        }
        protected void Page_Load(object sender, EventArgs e)
        {

        }
        private static void ShowAlertMessage(string error)
        {
            System.Web.UI.Page page = System.Web.HttpContext.Current.Handler as System.Web.UI.Page;
            if (page != null)
            {
                error = error.Replace("'", "\'");
                System.Web.UI.ScriptManager.RegisterStartupScript(page, page.GetType(), "err_msg", "alert('" + error + "');", true);
            }
        }
        private void LogError(Exception ex)
        {
            string logFilePath = Server.MapPath("~/ErrorLog.txt");
            using (StreamWriter writer = new StreamWriter(logFilePath, true))
            {
                writer.WriteLine("----------------------------------------------------------------------------------------------------------------------------");
                writer.WriteLine("Error Logged at " + DateTime.Now);
                writer.WriteLine("Error Message: " + ex.Message);
                writer.WriteLine("Stack Trace: " + ex.StackTrace);
            }
        }

        protected void btngo_Click(object sender, EventArgs e)
        {
            try
            {
                DateTime fromDate = DateTime.Parse(txtfrom.Text);
                DateTime toDate = DateTime.Parse(txtto.Text);
                DataTable dt = GetDataFromDatabase(fromDate, toDate);
                grdItemEntry.DataSource = dt;
                grdItemEntry.DataBind();
            }
            catch (Exception ex)
            {
                
            }
        }

        private DataTable GetDataFromDatabase(DateTime fromDate, DateTime toDate)
        {
            DataTable dt = new DataTable();

            try
            {
                CreateConnection();
                OpenConnection();

                _sqlCommand = new SqlCommand("Sp_GetItemTransactionsByDateRange", _sqlCommand.Connection);
                _sqlCommand.CommandType = CommandType.StoredProcedure;
                _sqlCommand.Parameters.AddWithValue("@FromDate", fromDate);
                _sqlCommand.Parameters.AddWithValue("@ToDate", toDate);

                using (SqlDataAdapter da = new SqlDataAdapter(_sqlCommand))
                {
                    da.Fill(dt);
                }
            }
            catch (Exception ex)
            {
               
            }
            finally
            {
             
            }

            return dt;
        }
    }
